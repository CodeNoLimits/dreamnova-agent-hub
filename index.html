check_agent() {
  local name=$1 url=$2 restart_cmd=$3
  if ! curl -sf "$url" > /dev/null 2>&1; then
    echo "[$(date)] $name DOWN ‚Äî restarting"
    eval "$restart_cmd"
    sleep 3
    if curl -sf "$url" > /dev/null 2>&1; then
      echo "[$(date)] $name RESTORED"
      notify "$name restored ‚úÖ"
    else
      echo "[$(date)] $name FAILED to restart"
      notify "$name failed to restart ‚ùå ‚Äî manual check needed"
    fi
    echo "### $(date '+%H:%M') ‚Äî AUTO-HEAL $name" >> $BRIDGE
  fi
}
